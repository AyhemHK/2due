//user table
CREATE TABLE users (
    id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    username varchar(255),
    email VARCHAR(50),
    password varchar(255),
    reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
)

CREATE TABLE board (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
user_id INT(6) UNSIGNED,
name TEXT(30),
archive int(3) UNSIGNED DEFAULT NULL,
)

CREATE TABLE board_id(
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
board_id INT(6) UNSIGNED,
task TEXT(48),
task_notes TEXT(64) DEFAULT NULL,
column int(3) UNSIGNED DEFAULT NULL,
)
